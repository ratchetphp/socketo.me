 14: /**
 15:  * Plugin class that will add request and response logging to an HTTP request.
 16:  *
 17:  * The log plugin can log contextual information about an HTTP transaction,
 18:  * request and response headers, request and response bodies, and cURL debug
 19:  * information.  These different logging levels can be customized using bitwise
 20:  * operators on LOG_CONTEXT (transaction information), LOG_HEADERS (request and
 21:  * response headers), LOG_BODY (request and response entity bodies), and
 22:  * LOG_DEBUG (cURL info).
 23:  *
 24:  * You will only receive valuable logging information from requests sent using
 25:  * cURL.  Requests with mock responses will not provide valuable information
 26:  * with the log plugin.
 27:  *
 28:  * Be careful when logging entity bodies; before they can
 29:  * be logged, the entire request and response entity bodies must be loaded into
 30:  * memory.
 31:  */

177:      *
178:      * @param RequestInterface $request Request to log
179:      * @param Response $response (optional) Response to log
180:      */
181:     private function log(RequestInterface $request, Response $response = null)
182:     {
183:         $message = '';
184: 
185:         if ($this->settings & self::LOG_CONTEXT) {
186:             // Log common contextual information
187:             $message = $request->getHost() . ' - "' .  $request->getMethod()
188:                 . ' ' . $request->getResourceUri() . ' '
189:                 . strtoupper($request->getScheme()) . '/'
190:                 . $request->getProtocolVersion() . '"';
191: 
192:             // If a response is set, then log additional contextual information
193:             if ($response) {
194:                 $message .= sprintf(' - %s %s - %s %s %s',
195:                     $response->getStatusCode(),
196:                     $response->getContentLength() ?: 0,
197:                     $response->getInfo('total_time'),
198:                     $response->getInfo('speed_upload'),
199:                     $response->getInfo('speed_download')
200:                 );
201:             }
202:         }
203: 
204:         // Check if we are logging anything that will come from cURL
205:         if ($request->getParams()->get('curl_handle') && ($this->settings & self::LOG_DEBUG || $this->settings & self::LOG_HEADERS || $this->settings & self::LOG_BODY)) {
206: 
207:             // If context logging too, then add a new line for cleaner messages
208:             if ($this->settings & self::LOG_CONTEXT) {
209:                 $message .= "\n";
210:             }
211: 
212:             // Filter cURL's verbose output based on config settings
213:             $message .= $this->parseCurlLog($request);
214: 
215:             // Log the response body if the response is available
216:             if ($this->settings & self::LOG_BODY && $response) {
217:                 if ($request->getParams()->get('response_wire')) {
218:                     $message .= (string) $request->getParams()->get('response_wire');
219:                 } else {
220:                     $message .= $response->getBody(true);
221:                 }
222:             }
223:         }
224: 
225:         // Send the log message to the adapter, adding a category and host
226:         $priority = $response && !$response->isSuccessful() ? LOG_ERR : LOG_DEBUG;
227:         $this->logAdapter->log(trim($message), $priority, array(
228:             'category' => 'guzzle.request',
229:             'host'     => $this->hostname
230:         ));
231:     }

235:      *
236:      * @param RequestInterface $request Request that has a curl handle
237:      *
238:      * @return string
239:      */
240:     protected function parseCurlLog(RequestInterface $request)
241:     {
242:         $message = '';
243:         $handle = $request->getParams()->get('curl_handle');
244:         $stderr = $handle->getStderr(true);
245:         if ($stderr) {
246:             rewind($stderr);
247:             $addedBody = false;
248:             while ($line = fgets($stderr)) {
249:                 // * - Debug | < - Downstream | > - Upstream
250:                 if ($line[0] == '*') {
251:                     if ($this->settings & self::LOG_DEBUG) {
252:                         $message .= $line;
253:                     }
254:                 } else if ($this->settings & self::LOG_HEADERS) {
255:                     $message .= $line;
256:                 }
257:                 // Add the request body if needed
258:                 if ($this->settings & self::LOG_BODY) {
259:                     if (trim($line) == '' && $request instanceof EntityEnclosingRequestInterface) {
260:                         if ($request->getParams()->get('request_wire')) {
261:                             $message .= (string) $request->getParams()->get('request_wire') . "\r\n";
262:                         } else {
263:                             $message .= (string) $request->getBody() . "\r\n";
264:                         }
265:                         $addedBody = true;
266:                     }
267:                 }
268:             }
269:         }
270:

272:     }
