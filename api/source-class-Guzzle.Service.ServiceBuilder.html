 32:      *
 33:      * @return ServiceBuilder
 34:      * @throws RuntimeException if a file cannot be openend
 35:      * @throws LogicException when trying to extend a missing client
 36:      */
 37:     public static function factory($data, $extension = null)
 38:     {
 39:         $config = array();
 40:         if (is_string($data)) {
 41:             if (!is_readable($data)) {
 42:                 throw new \RuntimeException('Unable to open ' . $data);
 43:             }
 44:             $extension = $extension ?: pathinfo($data, PATHINFO_EXTENSION);
 45:             if ($extension == 'xml') {
 46:                 $data = new \SimpleXMLElement($data, null, true);
 47:             } else if ($extension == 'js' || $extension == 'json') {
 48:                 $config = json_decode(file_get_contents($data), true);
 49:             } else {
 50:                 throw new \RuntimeException('Unknown file type ' . $extension);
 51:             }
 52:         } else if (is_array($data)) {
 53:             $config = $data;
 54:         } else if (!($data instanceof \SimpleXMLElement)) {
 55:             throw new \InvalidArgumentException('Must pass a file name, array, or SimpleXMLElement');
 56:         }
 57: 
 58:         if ($data instanceof \SimpleXMLElement) {
 59:             foreach ($data->clients->client as $client) {
 60:                 $row = array();
 61:                 foreach ($client->param as $param) {
 62:                     $row[(string) $param->attributes()->name] = (string) $param->attributes()->value;
 63:                 }
 64:                 $config[(string) $client->attributes()->name] = array(
 65:                     'class'   => (string) $client->attributes()->class,
 66:                     'extends' => (string) $client->attributes()->extends,
 67:                     'params'  => $row
 68:                 );
 69:             }
 70:         }
 71: 
 72:         // Validate the configuration and handle extensions
 73:         foreach ($config as $name => &$client) {
 74:             $client['params'] = isset($client['params']) ? $client['params'] : array();
 75:             // Check if this client builder extends another client
 76:             if (!empty($client['extends'])) {
 77:                 // Make sure that the service it's extending has been defined

 79:                     throw new \LogicException($name . ' is trying to extend a non-existent service: ' . $client['extends']);
 80:                 }

 82:                     ? $config[$client['extends']]['class'] : $client['class'];
 83:                 $client['params'] = array_merge($config[$client['extends']]['params'], $client['params']);
 84:             }
 85:             $client['class'] = str_replace('.', '\\', $client['class']);
 86:         }
 87: 
 88:         return new self($config);
 89:     }
