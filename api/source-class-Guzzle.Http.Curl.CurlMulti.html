 11: /**
 12:  * Send {@see RequestInterface} objects in parallel using curl_multi
 13:  *
 14:  * This implementation allows developers to build applications that require
 15:  * blocking calls as if they were using easy handles while still taking
 16:  * advantage of shared persistent connections and globally parallel requests.
 17:  *
 18:  * The implementation of the CurlMulti class allows callers to send blocking
 19:  * requests that return back to the caller when their requests complete--
 20:  * regardless of whether or not previously sending requests in the curl_multi
 21:  * object have completed.  The implementation relies on managing the scope in
 22:  * which a caller adds a request to the CurlMulti object, and tracking the
 23:  * requests in the current scope until they complete.  Although the CurlMulti
 24:  * object only tracks whether or not requests in the current scope have
 25:  * completed, it still sends all requests added to the object in parallel.

339:             while ($done = curl_multi_info_read($this->multiHandle)) {

466: 
467:         if ($request->getState() != RequestInterface::STATE_TRANSFER) {
468:             $this->remove($request);
469:         }
470:     }
