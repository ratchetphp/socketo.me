 97:     /**
 98:      * Create a new Response based on a raw response message
 99:      *
100:      * @param string $message Response message
101:      *
102:      * @return Response
103:      * @throws InvalidArgumentException if an empty $message is provided
104:      */
105:     public static function fromMessage($message)
106:     {
107:         if (!$message) {
108:             throw new InvalidArgumentException('No response message provided');
109:         }
110: 
111:         // Normalize line endings
112:         $message = preg_replace("/([^\r])(\n)\b/", "$1\r\n", $message);
113:         $protocol = $code = $status = '';
114:         $parts = explode("\r\n\r\n", $message, 2);
115:         $headers = new Collection();
116: 
117:         foreach (array_values(array_filter(explode("\r\n", $parts[0]))) as $i => $line) {
118:             // Remove newlines from headers
119:             $line = implode(' ', explode("\n", $line));
120:             if ($i === 0) {
121:                 // Check the status line
122:                 list($protocol, $code, $status) = array_map('trim', explode(' ', $line, 3));
123:             } else if (strpos($line, ':')) {
124:                 // Add a header
125:                 list($key, $value) = array_map('trim', explode(':', $line, 2));
126:                 // Headers are case insensitive
127:                 $headers->add($key, $value);
128:             }
129:         }
130: 
131:         $body = null;
132: 
133:         if (isset($parts[1]) && $parts[1] != "\n") {
134:             $body = EntityBody::factory(trim($parts[1]));
135:             // Always set the appropriate Content-Length if Content-Legnth
136:             $headers['Content-Length'] = $body->getSize();
137:         }
138: 
139:         $response = new static(trim($code), $headers, $body);
140:         $response->setProtocol($protocol)
141:                  ->setStatus($code, $status);
142: 
143:         return $response;
144:     }
