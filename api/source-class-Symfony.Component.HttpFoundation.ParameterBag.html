 92:      *
 93:      * @param string  $path    The key
 94:      * @param mixed   $default The default value if the parameter key does not exist
 95:      * @param boolean $deep If true, a path like foo[bar] will find deeper items
 96:      *
 97:      * @api
 98:      */
 99:     public function get($path, $default = null, $deep = false)
100:     {
101:         if (!$deep || false === $pos = strpos($path, '[')) {
102:             return array_key_exists($path, $this->parameters) ? $this->parameters[$path] : $default;
103:         }
104: 
105:         $root = substr($path, 0, $pos);
106:         if (!array_key_exists($root, $this->parameters)) {
107:             return $default;
108:         }
109: 
110:         $value = $this->parameters[$root];
111:         $currentKey = null;
112:         for ($i = $pos, $c = strlen($path); $i < $c; $i++) {
113:             $char = $path[$i];
114: 
115:             if ('[' === $char) {
116:                 if (null !== $currentKey) {
117:                     throw new \InvalidArgumentException(sprintf('Malformed path. Unexpected "[" at position %d.', $i));
118:                 }
119: 
120:                 $currentKey = '';
121:             } elseif (']' === $char) {
122:                 if (null === $currentKey) {
123:                     throw new \InvalidArgumentException(sprintf('Malformed path. Unexpected "]" at position %d.', $i));
124:                 }
125: 
126:                 if (!is_array($value) || !array_key_exists($currentKey, $value)) {
127:                     return $default;
128:                 }
129: 
130:                 $value = $value[$currentKey];
131:                 $currentKey = null;
132:             } else {
133:                 if (null === $currentKey) {
134:                     throw new \InvalidArgumentException(sprintf('Malformed path. Unexpected "%s" at position %d.', $char, $i));
135:                 }
136: 
137:                 $currentKey .= $char;
138:             }
139:         }
140: 
141:         if (null !== $currentKey) {
142:             throw new \InvalidArgumentException(sprintf('Malformed path. Path must end with "]".'));
143:         }
144: 
145:         return $value;
146:     }
