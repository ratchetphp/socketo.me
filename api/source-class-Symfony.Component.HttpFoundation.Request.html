 230:     /**
 231:      * Creates a Request based on a given URI and configuration.
 232:      *
 233:      * @param string $uri        The URI
 234:      * @param string $method     The HTTP method
 235:      * @param array  $parameters The request (GET) or query (POST) parameters
 236:      * @param array  $cookies    The request cookies ($_COOKIE)
 237:      * @param array  $files      The request files ($_FILES)
 238:      * @param array  $server     The server parameters ($_SERVER)
 239:      * @param string $content    The raw body data
 240:      *
 241:      * @return Request A Request instance
 242:      *
 243:      * @api
 244:      */
 245:     static public function create($uri, $method = 'GET', $parameters = array(), $cookies = array(), $files = array(), $server = array(), $content = null)
 246:     {
 247:         $defaults = array(
 248:             'SERVER_NAME'          => 'localhost',
 249:             'SERVER_PORT'          => 80,
 250:             'HTTP_HOST'            => 'localhost',
 251:             'HTTP_USER_AGENT'      => 'Symfony/2.X',
 252:             'HTTP_ACCEPT'          => 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
 253:             'HTTP_ACCEPT_LANGUAGE' => 'en-us,en;q=0.5',
 254:             'HTTP_ACCEPT_CHARSET'  => 'ISO-8859-1,utf-8;q=0.7,*;q=0.7',
 255:             'REMOTE_ADDR'          => '127.0.0.1',
 256:             'SCRIPT_NAME'          => '',
 257:             'SCRIPT_FILENAME'      => '',
 258:             'SERVER_PROTOCOL'      => 'HTTP/1.1',
 259:             'REQUEST_TIME'         => time(),
 260:         );
 261: 
 262:         $components = parse_url($uri);
 263:         if (isset($components['host'])) {
 264:             $defaults['SERVER_NAME'] = $components['host'];
 265:             $defaults['HTTP_HOST'] = $components['host'];
 266:         }
 267: 
 268:         if (isset($components['scheme'])) {
 269:             if ('https' === $components['scheme']) {
 270:                 $defaults['HTTPS'] = 'on';
 271:                 $defaults['SERVER_PORT'] = 443;
 272:             }
 273:         }
 274: 
 275:         if (isset($components['port'])) {
 276:             $defaults['SERVER_PORT'] = $components['port'];
 277:             $defaults['HTTP_HOST'] = $defaults['HTTP_HOST'].':'.$components['port'];
 278:         }
 279: 
 280:         if (isset($components['user'])) {
 281:             $defaults['PHP_AUTH_USER'] = $components['user'];
 282:         }
 283: 
 284:         if (isset($components['pass'])) {
 285:             $defaults['PHP_AUTH_PW'] = $components['pass'];
 286:         }
 287: 
 288:         if (!isset($components['path'])) {
 289:             $components['path'] = '';
 290:         }
 291: 
 292:         switch (strtoupper($method)) {
 293:             case 'POST':
 294:             case 'PUT':
 295:             case 'DELETE':
 296:                 $defaults['CONTENT_TYPE'] = 'application/x-www-form-urlencoded';
 297:             case 'PATCH':
 298:                 $request = $parameters;
 299:                 $query = array();
 300:                 break;
 301:             default:
 302:                 $request = array();
 303:                 $query = $parameters;
 304:                 break;
 305:         }
 306: 
 307:         if (isset($components['query'])) {
 308:             $queryString = html_entity_decode($components['query']);
 309:             parse_str($queryString, $qs);
 310:             if (is_array($qs)) {
 311:                 $query = array_replace($qs, $query);
 312:             }
 313:         }
 314:         $queryString = http_build_query($query);
 315: 
 316:         $uri = $components['path'].($queryString ? '?'.$queryString : '');
 317: 
 318:         $server = array_replace($defaults, $server, array(
 319:             'REQUEST_METHOD' => strtoupper($method),
 320:             'PATH_INFO'      => '',
 321:             'REQUEST_URI'    => $uri,
 322:             'QUERY_STRING'   => $queryString,
 323:         ));
 324: 
 325:         return new static($query, $request, array(), $cookies, $files, $server, $content);
 326:     }

 413:     {
 414:         $_GET = $this->query->all();

 416:         $_SERVER = $this->server->all();
 417:         $_COOKIE = $this->cookies->all();

1279:     /**
1280:      * Prepares the base URL.
1281:      *
1282:      * @return string
1283:      */
1284:     protected function prepareBaseUrl()
1285:     {
1286:         $filename = basename($this->server->get('SCRIPT_FILENAME'));
1287: 
1288:         if (basename($this->server->get('SCRIPT_NAME')) === $filename) {
1289:             $baseUrl = $this->server->get('SCRIPT_NAME');
1290:         } elseif (basename($this->server->get('PHP_SELF')) === $filename) {
1291:             $baseUrl = $this->server->get('PHP_SELF');
1292:         } elseif (basename($this->server->get('ORIG_SCRIPT_NAME')) === $filename) {
1293:             $baseUrl = $this->server->get('ORIG_SCRIPT_NAME'); // 1and1 shared hosting compatibility
1294:         } else {
1295:             // Backtrack up the script_filename to find the portion matching
1296:             // php_self
1297:             $path    = $this->server->get('PHP_SELF', '');
1298:             $file    = $this->server->get('SCRIPT_FILENAME', '');
1299:             $segs    = explode('/', trim($file, '/'));
1300:             $segs    = array_reverse($segs);
1301:             $index   = 0;
1302:             $last    = count($segs);
1303:             $baseUrl = '';
1304:             do {
1305:                 $seg     = $segs[$index];
1306:                 $baseUrl = '/'.$seg.$baseUrl;
1307:                 ++$index;
1308:             } while (($last > $index) && (false !== ($pos = strpos($path, $baseUrl))) && (0 != $pos));
1309:         }
1310: 
1311:         // Does the baseUrl have anything in common with the request_uri?
1312:         $requestUri = $this->getRequestUri();
1313: 
1314:         if ($baseUrl && false !== $prefix = $this->getUrlencodedPrefix($requestUri, $baseUrl)) {
1315:             // full $baseUrl matches
1316:             return $prefix;
1317:         }
1318: 
1319:         if ($baseUrl && false !== $prefix = $this->getUrlencodedPrefix($requestUri, dirname($baseUrl))) {
1320:             // directory portion of $baseUrl matches
1321:             return rtrim($prefix, '/');
1322:         }
1323: 
1324:         $truncatedRequestUri = $requestUri;
1325:         if (($pos = strpos($requestUri, '?')) !== false) {
1326:             $truncatedRequestUri = substr($requestUri, 0, $pos);
1327:         }
1328: 
1329:         $basename = basename($baseUrl);
1330:         if (empty($basename) || !strpos(rawurldecode($truncatedRequestUri), $basename)) {
1331:             // no match whatsoever; set it blank
1332:             return '';
1333:         }
1334: 
1335:         // If using mod_rewrite or ISAPI_Rewrite strip the script filename
1336:         // out of baseUrl. $pos !== 0 makes sure it is not matching a value
1337:         // from PATH_INFO or QUERY_STRING
1338:         if ((strlen($requestUri) >= strlen($baseUrl)) && ((false !== ($pos = strpos($requestUri, $baseUrl))) && ($pos !== 0))) {
1339:             $baseUrl = substr($requestUri, 0, $pos + strlen($baseUrl));
1340:         }
1341: 
1342:         return rtrim($baseUrl, '/');
1343:     }

1345:     /**

1349:      */

1351:     {
1352:         $filename = basename($this->server->get('SCRIPT_FILENAME'));
1353:         $baseUrl = $this->getBaseUrl();
1354:         if (empty($baseUrl)) {

1361:             $basePath = $baseUrl;
1362:         }
