 51:     /**
 52:      * Parse a cookie string as set in a Set-Cookie HTTP header and return data
 53:      * about the cookie
 54:      *
 55:      * @param string $cookie Cookie header value to parse
 56:      * @param RequestInterface $request (optional) Used to get path and domain
 57:      *      when not using Cookie-2
 58:      * @param bool $decode (optional) Set to FALSE to not urldecode the values
 59:      *
 60:      * @return array|bool Returns FALSE on failure or returns an array of
 61:      * arrays, with each of the sub arrays including:
 62:      *      domain  (string) - Domain of the cookie
 63:      *      path    (string) - Path of the cookie
 64:      *      cookies (array)  - Associative array of cookie names and values
 65:      *      max_age (int)    - Lifetime of the cookie in seconds
 66:      *      version (int)    - Version of the cookie specification. RFC 2965 is 1
 67:      *      secure  (bool)   - Whether or not this is a secure cookie
 68:      *      discard (bool)   - Whether or not this is a discardable cookie
 69:      *      custom (string)  - Custom cookie data array
 70:      *      comment (string) - How the cookie is intended to be used
 71:      *      comment_url (str)- URL that contains info on how it will be used
 72:      *      port (array|str) - Array of ports or null
 73:      *      http_only (bool) - HTTP only cookie
 74:      */
 75:     public static function parseCookie($cookie, RequestInterface $request = null, $decode = true)
 76:     {
 77:         if (!$cookie) {
 78:             return false;
 79:         }
 80: 
 81:         // Explode the cookie string using a series of semicolons
 82:         $pieces = array_filter(array_map('trim', explode(';', $cookie)));
 83: 
 84:         // The name of the cookie (first kvp) must include an equal sign.
 85:         if (!strpos($pieces[0], '=')) {
 86:             return false;
 87:         }
 88: 
 89:         // Map cookie values into snake_case values and use these regexp's to find data
 90:         $parts = array(
 91:             'domain' => '/^Domain$/i',
 92:             'path' => '/^Path$/i',
 93:             'max_age' => '/^Max(\-|_)Age$/i',
 94:             'expires' => '/^Expires$/i',
 95:             'version' => '/^Version$/i',
 96:             'secure' => '/^Secure$/i',
 97:             'port' => '/^Port$/i',
 98:             'discard' => '/^Discard$/i',
 99:             'comment' => '/^Comment$/i',
100:             'comment_url' => '/^Comment(\-|_)Url$/i',
101:             'http_only' => '/^HttpOnly$/i'
102:         );
103: 
104:         // Create the default return array
105:         $data = array_fill_keys(array_keys($parts), null);
106:         $data['cookies'] = array();
107:         $data['data'] = array();
108:         $data['path'] = '/';
109:         $data['http_only'] = false;
110: 
111:         // Add the default domain and path using the request reference
112:         // These values are overridden if using Set-Cookie2 cookie values.
113:         if ($request) {
114:             $data['domain'] = $request->getHost();
115:             $data['path'] = $request->getPath();
116:         }
117: 
118:         $foundCookies = $foundNonCookies = 0;
119: 
120:         // Add the cookie pieces into the parsed data array
121:         foreach ($pieces as $part) {
122: 
123:             $avPair = array_map('trim', explode('=', $part, 2));
124:             $key = $avPair[0];
125: 
126:             if (count($avPair) === 1) {
127:                 // Matches secure, httpOnly, etc...
128:                 $value = true;
129:             } else {
130:                 $value = isset($avPair[1]) ? $avPair[1] : '';
131:                 // Remove wrapping quotes
132:                 if (strpos($value, '"') === 0) {
133:                     $value = substr($value, 1);
134:                 }
135:                 if (substr($value, -1, 1) == '"') {
136:                     $value = substr($value, 0, -1);
137:                 }
138:             }
139: 
140:             // Check if the key is in the cookie pieces
141:             $found = false;
142:             $needle = strtolower($key);
143:             if ($foundCookies) {
144:                 foreach ($parts as $mapValue => $regex) {
145:                     if (preg_match($regex, $needle)) {
146:                         $key = $needle = $mapValue;
147:                         $found = true;
148:                         break;
149:                     }
150:                 }
151:             }
152: 
153:             // Decode the value if $decode is TRUE
154:             $value = ($decode && $value && !is_bool($value)) ? urldecode($value) : $value;
155: 
156:             if (0 == $foundCookies || !$found) {
157:                 // If cookies have not yet been retrieved, or this value was
158:                 // not found in the cookie pieces array, treat as a cookie
159:                 // IF non-cookies have been parsed, then this isn't a cookie,
160:                 // but it's cookie data.  Cookies must be first, followed by data.
161:                 if ($foundNonCookies == 0) {
162:                     $data['cookies'][] = $key . ($value ? ('=' . $value) : '');
163:                     $foundCookies++;
164:                 } else {
165:                     $data['data'][$key] = $value;
166:                 }
167:             } else {
168:                 if ($key == 'port') {
169:                     $value = array_map('trim', explode(',', $value));
170:                 }
171:                 $data[$key] = $value;
172:                 $foundNonCookies++;
173:             }
174:         }
175: 
176:         // Calculate the expires date
177:         if (!$data['expires'] && $data['max_age']) {
178:             $data['expires'] = time() + (int)$data['max_age'];
179:         }
180: 
181:         return $data;
182:     }

184:     /**
185:      * Add cookies to a request based on the destination of the request and
186:      * the cookies stored in the storage backend.  Any previously set cookies
187:      * will be removed.
188:      *
189:      * @param RequestInterface $request Request to add cookies to.  If the
190:      *      request object already has a cookie header, then no further cookies
191:      *      will be added.
192:      *
193:      * @return array Returns an array of the cookies that were added
194:      */
195:     public function addCookies(RequestInterface $request)
196:     {
197:         $request->removeHeader('Cookie');
198:         // Find cookies that match this request
199:         $cookies = $this->jar->getCookies($request->getHost(), $request->getPath());
200:         $match = false;
201: 
202:         if ($cookies) {
203:             foreach ($cookies as $cookie) {
204:                 $match = true;
205:                 // If a port restriction is set, validate the port
206:                 if (!empty($cookie['port'])) {
207:                     if (!in_array($request->getPort(), $cookie['port'])) {
208:                         $match = false;
209:                     }
210:                 }
211:                 // Validate the secure flag
212:                 if ($cookie['secure']) {
213:                     if ($request->getScheme() != 'https') {
214:                         $match = false;
215:                     }
216:                 }
217:                 // If this request is eligible for the cookie, then merge it in
218:                 if ($match) {
219:                     $request->addCookie($cookie['cookie'][0], isset($cookie['cookie'][1]) ? $cookie['cookie'][1] : null);
220:                 }
221:             }
222:         }
223: 
224:         return $match && $cookies ? $cookies : array();
225:     }

227:     /**
228:      * Extracts cookies from an HTTP Response object, looking for Set-Cookie:
229:      * and Set-Cookie2: headers and persists them to the cookie storage.
230:      *
231:      * @param Response $response
232:      */
233:     public function extractCookies(Response $response)
234:     {
235:         if (!$cookie = $response->getSetCookie()) {
236:             return array();
237:         }
238: 
239:         $cookieData = array();
240:         foreach ($cookie as $c) {
241:             $cdata = self::parseCookie($c, $response->getRequest());
242: 
243:             //@codeCoverageIgnoreStart
244:             if (!$cdata) {
245:                 continue;
246:             }
247:             //@codeCoverageIgnoreEnd
248: 
249:             $cookies = array();
250:             // Break up cookie v2 into multiple cookies
251:             if (count($cdata['cookies']) == 1) {
252:                 $cdata['cookie'] = explode('=', $cdata['cookies'][0], 2);
253:                 unset($cdata['cookies']);
254:                 $cookies = array($cdata);
255:             } else {
256:                 foreach ($cdata['cookies'] as $cookie) {
257:                     $row = $cdata;
258:                     unset($row['cookies']);
259:                     $row['cookie'] = explode('=', $cookie, 2);
260:                     $cookies[] = $row;
261:                 }
262:             }
263: 
264:             if (count($cookies)) {
265:                 foreach ($cookies as &$c) {
266:                     $this->jar->save($c);
267:                     $cookieData[] = $c;
268:                 }
269:             }
270:         }
271: 
272:         return $cookieData;
273:     }
