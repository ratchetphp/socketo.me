 13: /**
 14:  * Plugin to enable the caching of GET and HEAD requests.  Caching can be done
 15:  * on all requests passing through this plugin or only after retrieving
 16:  * resources with cacheable response headers.  This is a simple implementation
 17:  * of RFC 2616 and should be considered a private transparent proxy cache
 18:  * (authorization and private data can be cached).
 19:  *
 20:  * Various params can be set on a request to modify the behavior of the cache plugin:
 21:  * - cache.key_filter:      Filtered headers and query string parameters to remove from
 22:  *                          a request when creating a cache key
 23:  * - cache.filter_strategy: Method to invoke when determining if a request can be cached.
 24:  *                          Must accept a request object and return boolean.
 25:  */

 47:         'Connection', 'Keep-Alive', 'Proxy-Authenticate', 'Proxy-Authorization',
 48:         'TE', 'Trailers', 'Transfer-Encoding', 'Upgrade', 'Set-Cookie', 'Set-Cookie2'

 57:      *                                               Explicit Cache-Control or Expires headers override this value.

 77:     /**
 78:      * Calculate the hash key of a request object
 79:      *
 80:      * @param RequestInterface $request Request to hash
 81:      * @param string           $raw     Set to TRUE to retrieve the un-encoded string for debugging
 82:      *
 83:      * @return string
 84:      */
 85:     public function getCacheKey(RequestInterface $request, $raw = false)
 86:     {
 87:         // See if the key has already been calculated
 88:         $key = $request->getParams()->get('cache.key');
 89: 
 90:         // Always recalculate when using the raw option
 91:         if (!$key || $raw) {
 92: 
 93:             // Generate the start of the key
 94:             $key = $request->getMethod() . '_' . $request->getScheme() . '_' . $request->getHost() . $request->getPath();
 95:             $filterHeaders = array('Cache-Control');
 96:             $filterQuery = array();
 97: 
 98:             // Check to see how and if the key should be filtered
 99:             foreach (explode(';', $request->getParams()->get('cache.key_filter')) as $part) {
100:                 $pieces = array_map('trim', explode('=', $part));
101:                 if (isset($pieces[1])) {
102:                     $remove = array_map('trim', explode(',', $pieces[1]));
103:                     if ($pieces[0] == 'header') {
104:                         $filterHeaders = array_merge($filterHeaders, $remove);
105:                     } elseif ($pieces[0] == 'query') {
106:                         $filterQuery = array_merge($filterQuery, $remove);
107:                     }
108:                 }
109:             }
110: 
111:             // Use the filtered query string
112:             $queryString = (string) $request->getQuery()->filter(function($key, $value) use ($filterQuery) {
113:                 return !in_array($key, $filterQuery);
114:             });
115: 
116:             // Use the filtered headers
117:             $headerString = http_build_query($request->getHeaders()->map(function($key, $value) {
118:                 return count($value) == 1 ? $value[0] : $value;
119:             })->filter(function($key, $value) use ($filterHeaders) {
120:                 return !in_array($key, $filterHeaders);
121:             })->getAll());
122: 
123:             if ($raw) {
124:                 $key = strtolower('gz_' . $key . $queryString . '_' . $headerString);
125:             } else {
126:                 $key = strtolower('gz_' . md5($key . $queryString . '_' . $headerString));
127:                 $request->getParams()->set('cache.key', $key);
128:             }
129:         }
130: 
131:         return $key;
132:     }

134:     /**
135:      * Check if a response in cache will satisfy the request before sending
136:      *
137:      * @param Event $event
138:      */
139:     public function onRequestBeforeSend(Event $event)
140:     {
141:         $request = $event['request'];
142: 
143:         // Only cache cacheable requests
144:         if ($cacheFilter = $request->getParams()->get('cache.filter_strategy')) {
145:             if (!call_user_func($cacheFilter, $request)) {
146:                 return;
147:             }
148:         } elseif (!$request->canCache()) {
149:             return;
150:         }
151: 
152:         $hashKey = $this->getCacheKey($request);
153:         $this->cached[$request] = $hashKey;
154:         $cachedData = $this->adapter->fetch($hashKey);
155: 
156:         // If the cached data was found, then make the request into a
157:         // manually set request
158:         if ($cachedData) {
159:             unset($this->cached[$request]);
160:             $response = new Response($cachedData[0], $cachedData[1], $cachedData[2]);
161:             $response->setHeader('Age', time() - strtotime($response->getDate() ?: 'now'));
162:             if (!$response->hasHeader('X-Guzzle-Cache')) {
163:                 $response->setHeader('X-Guzzle-Cache', "key={$hashKey}");
164:             }
165: 
166:             // Validate that the response satisfies the request
167:             if ($this->canResponseSatisfyRequest($request, $response)) {
168:                 $request->setResponse($response);
169:             }
170:         }
171:     }

174:      * If possible, store a response in cache after sending
175:      *
176:      * @param Event $event
177:      */
178:     public function onRequestSent(Event $event)
179:     {
180:         $request = $event['request'];
181:         $response = $event['response'];
182: 
183:         if ($response->canCache() && isset($this->cached[$request])) {
184: 
185:             $cacheKey = $this->cached[$request];
186:             unset($this->cached[$request]);
187: 
188:             if ($response->isSuccessful()) {
189:                 if ($request->getParams()->get('cache.override_ttl')) {
190:                     $lifetime = $request->getParams()->get('cache.override_ttl');
191:                     $response->setHeader('X-Guzzle-Cache', "key={$cacheKey}, ttl={$lifetime}");
192:                 } else {
193:                     $lifetime = $response->getMaxAge();
194:                 }
195:                 $this->saveCache($cacheKey, $response, $lifetime);
196:             }

200:     /**
201:      * Revalidate a cached response
202:      *
203:      * @param RequestInterface $request  Request to revalidate
204:      * @param Response         $response Response to revalidate
205:      *
206:      * @return bool
207:      */
208:     public function revalidate(RequestInterface $request, Response $response)
209:     {
210:         static $replaceHeaders = array('Date', 'Expires', 'Cache-Control', 'ETag', 'Last-Modified');
211: 
212:         $revalidate = clone $request;
213:         $revalidate->getEventDispatcher()->removeSubscriber($this);
214:         $revalidate->removeHeader('Pragma')
215:                    ->removeHeader('Cache-Control')
216:                    ->setHeader('If-Modified-Since', $response->getDate());
217: 
218:         if ($response->getEtag()) {
219:             $revalidate->setHeader('If-None-Match', '"' . $response->getEtag() . '"');
220:         }
221: 
222:         try {
223: 
224:             $validateResponse = $revalidate->send();
225:             if ($validateResponse->getStatusCode() == 200) {
226:                 // The server does not support validation, so use this response
227:                 $request->setResponse($validateResponse);
228:                 // Store this response in cache if possible
229:                 if ($validateResponse->canCache()) {
230:                     $this->saveCache($this->getCacheKey($request), $validateResponse, $validateResponse->getMaxAge());
231:                 }
232: 
233:                 return false;
234:             }
235: 
236:             if ($validateResponse->getStatusCode() == 304) {
237:                 // Make sure that this response has the same ETage
238:                 if ($validateResponse->getEtag() != $response->getEtag()) {
239:                     return false;
240:                 }
241:                 // Replace cached headers with any of these headers from the
242:                 // origin server that might be more up to date
243:                 $modified = false;
244:                 foreach ($replaceHeaders as $name) {
245:                     if ($validateResponse->hasHeader($name)) {
246:                         $modified = true;
247:                         $response->setHeader($name, $validateResponse->getHeader($name));
248:                     }
249:                 }
250:                 // Store the updated response in cache
251:                 if ($modified && $response->canCache()) {
252:                     $this->saveCache($this->getCacheKey($request), $response, $response->getMaxAge());
253:                 }
254: 
255:                 return true;
256:             }
257: 
258:         } catch (BadResponseException $e) {
259: 
260:             // 404 errors mean the resource no longer exists, so remove from
261:             // cache, and prevent an additional request by throwing the exception
262:             if ($e->getResponse()->getStatusCode() == 404) {
263:                 $this->adapter->delete($this->getCacheKey($request));
264:                 throw $e;
265:             }
266:         }
267: 
268:         // Other exceptions encountered in the revalidation request are ignored
269:         // in hopes that sending a request to the origin server will fix it
270:         return false;
271:     }

273:     /**
274:      * Check if a cache response satisfies a request's caching constraints
275:      *
276:      * @param RequestInterface $request  Request to validate
277:      * @param Response         $response Response to validate
278:      *
279:      * @return bool
280:      */
281:     public function canResponseSatisfyRequest(RequestInterface $request, Response $response)
282:     {
283:         $responseAge = $response->getAge();
284: 
285:         // Check the request's max-age header against the age of the response
286:         if ($request->hasCacheControlDirective('max-age') &&
287:             $responseAge > $request->getCacheControlDirective('max-age')) {
288:             return false;
289:         }
290: 
291:         // Check the response's max-age header
292:         if ($response->isFresh() === false) {
293:             $maxStale = $request->getCacheControlDirective('max-stale');
294:             if (null !== $maxStale) {
295:                 if ($maxStale !== true && $response->getFreshness() < (-1 * $maxStale)) {
296:                     return false;
297:                 }
298:             } elseif ($responseAge > $response->getCacheControlDirective('max-age')) {
299:                 return false;
300:             }
301:         }
302: 
303:         // Only revalidate GET requests
304:         if ($request->getMethod() == RequestInterface::GET) {
305:             // Check if the response must be validated against the origin server
306:             if ($request->getHeader('Pragma') == 'no-cache' ||
307:                 $request->hasCacheControlDirective('no-cache') ||
308:                 $request->hasCacheControlDirective('must-revalidate') ||
309:                 $response->hasCacheControlDirective('must-revalidate') ||
310:                 $response->hasCacheControlDirective('no-cache')) {
311:                 // no-cache: When no parameters are present, always revalidate
312:                 // When parameters are present in no-cache and the request includes
313:                 // those same parameters, then the response must re-validate
314:                 // I'll need an example of what fields look like in order to
315:                 // implement a smarter version of no-cache
316: 
317:                 // Requests can decline to revalidate against the origin server
318:                 // by setting the cache.revalidate param to one of:
319:                 //      never  - To never revalidate and always contact the origin server
320:                 //      skip   - To skip revalidation and just use what is in cache
321:                 switch ($request->getParams()->get('cache.revalidate')) {
322:                     case 'never':
323:                         return false;
324:                     case 'skip':
325:                         return true;
326:                     default:
327:                         return $this->revalidate($request, $response);
328:                 }
329:             }
330:         }
331: 
332:         return true;
333:     }

335:     /**
336:      * Save data to the cache adapter
337:      *
338:      * @param string   $key      The cache key
339:      * @param Response $response The response to cache
340:      * @param int      $lifetime Amount of seconds to cache
341:      *
342:      * @return int Returns the lifetime of the cached data
343:      */
344:     protected function saveCache($key, Response $response, $lifetime = null)
345:     {
346:         // If the data is cacheable, then save it to the cache adapter
347:         if ($lifetime = $lifetime ?: $this->defaultLifetime) {
348:             // Remove excluded headers from the response  (see RFC 2616:13.5.1)
349:             foreach ($this->excludeResponseHeaders as $header) {
350:                 $response->removeHeader($header);
351:             }
352:             // Add a Date header to the response if none is set (for validation)
353:             if (!$response->getDate()) {
354:                 $response->setHeader('Date', Utils::getHttpDate('now'));
355:             }
356:             $this->adapter->save(
357:                 $key,
358:                 array($response->getStatusCode(), $response->getHeaders(), $response->getBody(true)),
359:                 $lifetime
360:             );
361:         }
362: 
363:         return $lifetime;
364:     }
