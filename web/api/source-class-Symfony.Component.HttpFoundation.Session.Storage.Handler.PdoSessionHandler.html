 23:      * @var \PDO PDO instance.
 24:      */

 28:      * @var array Database options.
 29:      */
 30:     private $dbOptions;
 31: 
 32:     /**
 33:      * Constructor.
 34:      *
 35:      * List of available options:
 36:      *  * db_table: The name of the table [required]
 37:      *  * db_id_col: The column where to store the session id [default: sess_id]
 38:      *  * db_data_col: The column where to store the session data [default: sess_data]
 39:      *  * db_time_col: The column where to store the timestamp [default: sess_time]
 40:      *
 41:      * @param \PDO  $pdo       A \PDO instance
 42:      * @param array $dbOptions An associative array of DB options
 43:      *
 44:      * @throws \InvalidArgumentException When "db_table" option is not provided
 45:      */
 46:     public function __construct(\PDO $pdo, array $dbOptions = array())
 47:     {
 48:         if (!array_key_exists('db_table', $dbOptions)) {
 49:             throw new \InvalidArgumentException('You must provide the "db_table" option for a PdoSessionStorage.');
 50:         }
 51:         if (\PDO::ERRMODE_EXCEPTION !== $pdo->getAttribute(\PDO::ATTR_ERRMODE)) {
 52:             throw new \InvalidArgumentException(sprintf('"%s" requires PDO error mode attribute be set to throw Exceptions (i.e. $pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION))', __CLASS__));
 53:         }
 54:         $this->pdo = $pdo;
 55:         $this->dbOptions = array_merge(array(
 56:             'db_id_col'   => 'sess_id',
 57:             'db_data_col' => 'sess_data',
 58:             'db_time_col' => 'sess_time',
 59:         ), $dbOptions);
 60:     }

 79:      * {@inheritDoc}
 80:      */
 81:     public function destroy($id)
 82:     {
 83:         // get table/column
 84:         $dbTable = $this->dbOptions['db_table'];
 85:         $dbIdCol = $this->dbOptions['db_id_col'];
 86: 
 87:         // delete the record associated with this id
 88:         $sql = "DELETE FROM $dbTable WHERE $dbIdCol = :id";
 89: 
 90:         try {
 91:             $stmt = $this->pdo->prepare($sql);
 92:             $stmt->bindParam(':id', $id, \PDO::PARAM_STR);
 93:             $stmt->execute();
 94:         } catch (\PDOException $e) {
 95:             throw new \RuntimeException(sprintf('PDOException was thrown when trying to manipulate session data: %s', $e->getMessage()), 0, $e);
 96:         }
 97: 
 98:         return true;
 99:     }

102:      * {@inheritDoc}
103:      */
104:     public function gc($lifetime)
105:     {
106:         // get table/column
107:         $dbTable   = $this->dbOptions['db_table'];
108:         $dbTimeCol = $this->dbOptions['db_time_col'];
109: 
110:         // delete the session records that have expired
111:         $sql = "DELETE FROM $dbTable WHERE $dbTimeCol < :time";
112:

114:             $stmt = $this->pdo->prepare($sql);
115:             $stmt->bindValue(':time', time() - $lifetime, \PDO::PARAM_INT);
116:             $stmt->execute();
117:         } catch (\PDOException $e) {
118:             throw new \RuntimeException(sprintf('PDOException was thrown when trying to manipulate session data: %s', $e->getMessage()), 0, $e);
119:         }

124:     /**
125:      * {@inheritDoc}
126:      */
127:     public function read($id)
128:     {
129:         // get table/columns
130:         $dbTable   = $this->dbOptions['db_table'];
131:         $dbDataCol = $this->dbOptions['db_data_col'];
132:         $dbIdCol   = $this->dbOptions['db_id_col'];
133: 
134:         try {
135:             $sql = "SELECT $dbDataCol FROM $dbTable WHERE $dbIdCol = :id";
136: 
137:             $stmt = $this->pdo->prepare($sql);
138:             $stmt->bindParam(':id', $id, \PDO::PARAM_STR);
139: 
140:             $stmt->execute();
141:             // it is recommended to use fetchAll so that PDO can close the DB cursor
142:             // we anyway expect either no rows, or one row with one column. fetchColumn, seems to be buggy #4777
143:             $sessionRows = $stmt->fetchAll(\PDO::FETCH_NUM);
144: 
145:             if (count($sessionRows) == 1) {
146:                 return base64_decode($sessionRows[0][0]);
147:             }
148: 
149:             // session does not exist, create it
150:             $this->createNewSession($id);
151: 
152:             return '';
153:         } catch (\PDOException $e) {
154:             throw new \RuntimeException(sprintf('PDOException was thrown when trying to read the session data: %s', $e->getMessage()), 0, $e);
155:         }
156:     }
157: 
158:     /**
159:      * {@inheritDoc}
160:      */
161:     public function write($id, $data)
162:     {
163:         // get table/column
164:         $dbTable   = $this->dbOptions['db_table'];
165:         $dbDataCol = $this->dbOptions['db_data_col'];
166:         $dbIdCol   = $this->dbOptions['db_id_col'];
167:         $dbTimeCol = $this->dbOptions['db_time_col'];
168: 
169:         //session data can contain non binary safe characters so we need to encode it
170:         $encoded = base64_encode($data);
171: 
172:         try {
173:             $driver = $this->pdo->getAttribute(\PDO::ATTR_DRIVER_NAME);
174: 
175:             if ('mysql' === $driver) {
176:                 // MySQL would report $stmt->rowCount() = 0 on UPDATE when the data is left unchanged
177:                 // it could result in calling createNewSession() whereas the session already exists in
178:                 // the DB which would fail as the id is unique
179:                 $stmt = $this->pdo->prepare(
180:                     "INSERT INTO $dbTable ($dbIdCol, $dbDataCol, $dbTimeCol) VALUES (:id, :data, :time) " .
181:                     "ON DUPLICATE KEY UPDATE $dbDataCol = VALUES($dbDataCol), $dbTimeCol = VALUES($dbTimeCol)"
182:                 );
183:                 $stmt->bindParam(':id', $id, \PDO::PARAM_STR);
184:                 $stmt->bindParam(':data', $encoded, \PDO::PARAM_STR);
185:                 $stmt->bindValue(':time', time(), \PDO::PARAM_INT);
186:                 $stmt->execute();
187:             } elseif ('oci' === $driver) {
188:                 $stmt = $this->pdo->prepare("MERGE INTO $dbTable USING DUAL ON($dbIdCol = :id) ".
189:                        "WHEN NOT MATCHED THEN INSERT ($dbIdCol, $dbDataCol, $dbTimeCol) VALUES (:id, :data, sysdate) " .
190:                        "WHEN MATCHED THEN UPDATE SET $dbDataCol = :data WHERE $dbIdCol = :id");
191: 
192:                 $stmt->bindParam(':id', $id, \PDO::PARAM_STR);
193:                 $stmt->bindParam(':data', $encoded, \PDO::PARAM_STR);
194:                 $stmt->execute();
195:             } else {
196:                 $stmt = $this->pdo->prepare("UPDATE $dbTable SET $dbDataCol = :data, $dbTimeCol = :time WHERE $dbIdCol = :id");
197:                 $stmt->bindParam(':id', $id, \PDO::PARAM_STR);
198:                 $stmt->bindParam(':data', $encoded, \PDO::PARAM_STR);
199:                 $stmt->bindValue(':time', time(), \PDO::PARAM_INT);
200:                 $stmt->execute();
201: 
202:                 if (!$stmt->rowCount()) {
203:                     // No session exists in the database to update. This happens when we have called
204:                     // session_regenerate_id()
205:                     $this->createNewSession($id, $data);
206:                 }
207:             }
208:         } catch (\PDOException $e) {
209:                 throw new \RuntimeException(sprintf('PDOException was thrown when trying to write the session data: %s', $e->getMessage()), 0, $e);
210:         }
211: 
212:         return true;
213:     }

225:         // get table/column
226:         $dbTable   = $this->dbOptions['db_table'];

237:         $stmt->bindParam(':data', $encoded, \PDO::PARAM_STR);
