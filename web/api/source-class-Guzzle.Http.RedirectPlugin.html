 15:  * Plugin to implement HTTP redirects. Can redirect like a web browser or using strict RFC 2616 compliance
 16:  */
 17: class RedirectPlugin implements EventSubscriberInterface
 18: {
 19:     const REDIRECT_COUNT = 'redirect.count';
 20:     const MAX_REDIRECTS = 'redirect.max';
 21:     const STRICT_REDIRECTS = 'redirect.strict';
 22:     const PARENT_REQUEST = 'redirect.parent_request';
 23:     const DISABLE = 'redirect.disable';

 26:      * @var int Default number of redirects allowed when no setting is supplied by a request
 27:      */

 31:      * {@inheritdoc}
 32:      */

 37:             'request.clone' => 'onRequestClone'

 51:     /**
 52:      * Called when a request receives a redirect response
 53:      *
 54:      * @param Event $event Event emitted
 55:      */
 56:     public function onRequestSent(Event $event)
 57:     {
 58:         $response = $event['response'];
 59:         $request = $event['request'];
 60: 
 61:         // Only act on redirect requests with Location headers
 62:         if (!$response || !$response->isRedirect() || !$response->hasHeader('Location')
 63:             || $request->getParams()->get(self::DISABLE)
 64:         ) {
 65:             return;
 66:         }
 67: 
 68:         // Prepare the request for a redirect and grab the original request that started the transaction
 69:         $originalRequest = $this->prepareRedirection($request);
 70: 
 71:         // Create a redirect request based on the redirect rules set on the request
 72:         $redirectRequest = $this->createRedirectRequest(
 73:             $request,
 74:             $event['response']->getStatusCode(),
 75:             trim($response->getHeader('Location')),
 76:             $originalRequest
 77:         );
 78: 
 79:         // Send the redirect request and hijack the response of the original request
 80:         $redirectResponse = $redirectRequest->send();
 81:         $redirectResponse->setPreviousResponse($event['response']);
 82:         $request->setResponse($redirectResponse);
 83:     }

 85:     /**
 86:      * Create a redirect request for a specific request object
 87:      *
 88:      * Takes into account strict RFC compliant redirection (e.g. redirect POST with POST) vs doing what most clients do
 89:      * (e.g. redirect POST with GET).
 90:      *
 91:      * @param RequestInterface $request    Request being redirected
 92:      * @param RequestInterface $original   Original request
 93:      * @param int              $statusCode Status code of the redirect
 94:      * @param string           $location   Location header of the redirect
 95:      *
 96:      * @return RequestInterface Returns a new redirect request
 97:      * @throws CouldNotRewindStreamException If the body needs to be rewound but cannot
 98:      */
 99:     protected function createRedirectRequest(
100:         RequestInterface $request,
101:         $statusCode,
102:         $location,
103:         RequestInterface $original
104:     ) {
105:         $redirectRequest = null;
106:         $strict = $original->getParams()->get(self::STRICT_REDIRECTS);
107:         // Use a GET request if this is an entity enclosing request and we are not forcing RFC compliance, but rather
108:         // emulating what all browsers would do
109:         if ($request instanceof EntityEnclosingRequestInterface && !$strict && $statusCode <= 302) {
110:             $redirectRequest = $this->cloneRequestWithGetMethod($request);
111:         } else {
112:             $redirectRequest = clone $request;
113:         }
114: 
115:         $location = Url::factory($location);
116:         // If the location is not absolute, then combine it with the original URL
117:         if (!$location->isAbsolute()) {
118:             $originalUrl = $redirectRequest->getUrl(true);
119:             // Remove query string parameters and just take what is present on the redirect Location header
120:             $originalUrl->getQuery()->clear();
121:             $location = $originalUrl->combine((string) $location);
122:         }
123: 
124:         $redirectRequest->setUrl($location);
125:         $redirectRequest->getParams()->set(self::PARENT_REQUEST, $request);
126: 
127:         // Rewind the entity body of the request if needed
128:         if ($redirectRequest instanceof EntityEnclosingRequestInterface && $redirectRequest->getBody()) {
129:             $body = $redirectRequest->getBody();
130:             // Only rewind the body if some of it has been read already, and throw an exception if the rewind fails
131:             if ($body->ftell() && !$body->rewind()) {
132:                 throw new CouldNotRewindStreamException(
133:                     'Unable to rewind the non-seekable entity body of the request after redirecting. cURL probably '
134:                     . 'sent part of body before the redirect occurred. Try adding acustom rewind function using on the '
135:                     . 'entity body of the request using setRewindFunction().'
136:                 );
137:             }
138:         }
139: 
140:         return $redirectRequest;
141:     }

143:     /**
144:      * Clone a request while changing the method to GET. Emulates the behavior of
145:      * {@see Guzzle\Http\Message\Request::clone}, but can change the HTTP method.
146:      *
147:      * @param EntityEnclosingRequestInterface $request Request to clone
148:      *
149:      * @return RequestInterface Returns a GET request
150:      */
151:     protected function cloneRequestWithGetMethod(EntityEnclosingRequestInterface $request)
152:     {
153:         // Create a new GET request using the original request's URL
154:         $redirectRequest = $request->getClient()->get($request->getUrl());
155:         $redirectRequest->getCurlOptions()->replace($request->getCurlOptions()->getAll());
156:         // Copy over the headers, while ensuring that the Content-Length is not copied
157:         $redirectRequest->setHeaders($request->getHeaders()->getAll())->removeHeader('Content-Length');
158:         $redirectRequest->setEventDispatcher(clone $request->getEventDispatcher());
159:         $redirectRequest->getParams()
160:             ->replace($request->getParams()->getAll())
161:             ->remove('curl_handle')->remove('queued_response')->remove('curl_multi');
162: 
163:         return $redirectRequest;
164:     }

166:     /**
167:      * Prepare the request for redirection and enforce the maximum number of allowed redirects per client
168:      *
169:      * @param RequestInterface $request Request to prepare and validate
170:      *
171:      * @return RequestInterface Returns the original request
172:      */
173:     protected function prepareRedirection(RequestInterface $request)
174:     {
175:         $original = $request;
176:         // The number of redirects is held on the original request, so determine which request that is
177:         while ($parent = $original->getParams()->get(self::PARENT_REQUEST)) {
178:             $original = $parent;
179:         }
180: 
181:         // Always associate the parent response with the current response so that a chain can be established
182:         if ($parent = $request->getParams()->get(self::PARENT_REQUEST)) {
183:             $request->getResponse()->setPreviousResponse($parent->getResponse());
184:         }
185: 
186:         $params = $original->getParams();
187:         // This is a new redirect, so increment the redirect counter
188:         $current = $params->get(self::REDIRECT_COUNT) + 1;
189:         $params->set(self::REDIRECT_COUNT, $current);
190: 
191:         // Use a provided maximum value or default to a max redirect count of 5
192:         $max = $params->hasKey(self::MAX_REDIRECTS)
193:             ? $params->get(self::MAX_REDIRECTS)
194:             : $this->defaultMaxRedirects;
195: 
196:         // Throw an exception if the redirect count is exceeded
197:         if ($current > $max) {
198:             return $this->throwTooManyRedirectsException($request);
199:         }
200: 
201:         return $original;
202:     }

204:     /**
205:      * Throw a too many redirects exception for a request
206:      *
207:      * @param RequestInterface $request Request
208:      * @throws TooManyRedirectsException when too many redirects have been issued
209:      */
210:     protected function throwTooManyRedirectsException(RequestInterface $request)
211:     {
212:         $responses = array();
213: 
214:         // Create a nice message to use when throwing the exception that shows each request/response transaction
215:         do {
216:             $response = $request->getResponse();
217:             $responses[] = '> ' . $request->getRawHeaders() . "\n\n< " . $response->getRawHeaders();
218:             $request = $response->getPreviousResponse() ? $response->getPreviousResponse()->getRequest() : null;
219:         } while ($request);
220: 
221:         $transaction = implode("* Sending redirect request\n", array_reverse($responses));
222: 
223:         throw new TooManyRedirectsException("Too many redirects were issued for this transaction:\n{$transaction}");
224:     }
