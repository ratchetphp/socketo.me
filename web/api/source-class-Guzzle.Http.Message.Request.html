484:     {

504:                 'line'          => $data,
505:                 'status_code'   => $code,

715:     /**
716:      * Process a received response
717:      *
718:      * @throws BadResponseException on unsuccessful responses
719:      */
720:     protected function processResponse()
721:     {
722:         // Use the queued response if one is set
723:         if ($this->getParams()->get('queued_response')) {
724:             $this->response = $this->getParams()->get('queued_response');
725:             $this->responseBody = $this->response->getBody();
726:             $this->getParams()->remove('queued_response');
727:         } elseif (!$this->response) {
728:             // If no response, then processResponse shouldn't have been called
729:             $e = new RequestException('Error completing request');
730:             $e->setRequest($this);
731:             throw $e;
732:         }
733: 
734:         $this->state = self::STATE_COMPLETE;
735: 
736:         // A request was sent, but we don't know if we'll send more or if the final response will be successful
737:         $this->dispatch('request.sent', $this->getEventArray());
738: 
739:         // Some response processors will remove the response or reset the state (example: ExponentialBackoffPlugin)
740:         if ($this->state == RequestInterface::STATE_COMPLETE) {
741: 
742:             // The request completed, so the HTTP transaction is complete
743:             $this->dispatch('request.complete', $this->getEventArray());
744: 
745:             // If the response is bad, allow listeners to modify it or throw exceptions. You can change the response by
746:             // modifying the Event object in your listeners or calling setResponse() on the request
747:             if ($this->response->isError()) {
748:                 $event = new Event($this->getEventArray());
749:                 $this->getEventDispatcher()->dispatch('request.error', $event);
750:                 // Allow events of request.error to quietly change the response
751:                 if ($event['response'] !== $this->response) {
752:                     $this->response = $event['response'];
753:                 }
754:             }
755: 
756:             // If a successful response was received, dispatch an event
